NG-HOLISTIC / FORMS

Set of angular libraries which provide a way to relaiable build complex forms with minimal effort via declaration in js objects. Every field of the form defined as a simple js object via simple rules. Here example of the form with text input field.

Code

    
        @Component({
            selector: 'hlc-sample-form',
            template: '<hlc-clr-form [group]="group"></hlc-clr-form>'
        })
        export class SampleFormComponent {
            readonly group: ClrFormLayouts.ClrFormLayout = {
                kind: 'fields',
                fields: [{ id: 'name', kind: 'TextField', props: { label: 'Name' } }]
            }
        }
